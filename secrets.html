#include <stdio.h>

int main() {
    int a = 5, b = 10;
    int *p1, *p2, **pp1;

    p1 = &a;    
    p2 = &b;    

    pp1 = &p1;  
    
    **pp1 += *p2;   

    p1 = p2;    

    *p1 += 3;   

    p2 = &a;    
    
    *p2 -= 2;   

    int c = *p1 + **pp1;  

    printf("Der Wert von c ist: %d\n", c);

    return 0;
}
